STEP 2 — CREA LE TABELLE (SQL Editor)
───────────────────────────────────────

Copia e incolla TUTTO questo blocco SQL e clicca "Run":

------------------------------------------------------------
-- TABELLA PROFILI (estende auth.users)
create table public.profiles (
  id uuid references auth.users on delete cascade primary key,
  full_name text,
  email text,
  role text default 'athlete' check (role in ('admin','athlete')),
  coach_id uuid references public.profiles(id),
  program_id text,
  current_week integer default 1,
  created_at timestamptz default now()
);

-- Trigger: crea profilo automaticamente al signup
create or replace function public.handle_new_user()
returns trigger as $$
begin
  insert into public.profiles (id, email)
  values (new.id, new.email);
  return new;
end;
$$ language plpgsql security definer;

create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();

-- TABELLA LOGBOOK
create table public.logbook (
  id bigserial primary key,
  athlete_id uuid references public.profiles(id) on delete cascade,
  key text not null,
  s1_kg numeric, s1_reps integer,
  s2_kg numeric, s2_reps integer,
  done boolean default false,
  updated_at timestamptz default now(),
  unique(athlete_id, key)
);

-- TABELLA MESSAGGI
create table public.messages (
  id bigserial primary key,
  athlete_id uuid references public.profiles(id),
  recipient_id uuid references public.profiles(id),
  sender_role text check (sender_role in ('admin','athlete')),
  text text not null,
  created_at timestamptz default now()
);

-- TABELLA PROGRAMMI (per blocchi personalizzati futuri)
create table public.programs (
  id text primary key,
  name text,
  data jsonb,
  created_by uuid references public.profiles(id),
  created_at timestamptz default now()
);

-- ROW LEVEL SECURITY
alter table public.profiles enable row level security;
alter table public.logbook enable row level security;
alter table public.messages enable row level security;
alter table public.programs enable row level security;

-- Policies: ogni utente vede solo i suoi dati
create policy "Profilo proprio" on public.profiles
  for all using (auth.uid() = id);

create policy "Admin vede tutti i profili" on public.profiles
  for select using (
    exists (select 1 from public.profiles where id = auth.uid() and role = 'admin')
  );

create policy "Admin aggiorna profili" on public.profiles
  for update using (
    exists (select 1 from public.profiles where id = auth.uid() and role = 'admin')
  );

create policy "Logbook proprio" on public.logbook
  for all using (athlete_id = auth.uid());

create policy "Admin vede logbook atleti" on public.logbook
  for select using (
    exists (select 1 from public.profiles where id = auth.uid() and role = 'admin')
  );

create policy "Messaggi propri" on public.messages
  for all using (athlete_id = auth.uid() or recipient_id = auth.uid());

create policy "Admin vede tutti i messaggi" on public.messages
  for all using (
    exists (select 1 from public.profiles where id = auth.uid() and role = 'admin')
  );

create policy "Programmi pubblici" on public.programs
  for select using (true);

create policy "Admin gestisce programmi" on public.programs
  for all using (
    exists (select 1 from public.profiles where id = auth.uid() and role = 'admin')
  );
